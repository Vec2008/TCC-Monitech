Arduino IDE — Dispositivo Monitech com WebSocket Server e Medição PZEM-004T 
#include <WiFi.h>
#include <WebSocketsServer.h>
#include <ArduinoJson.h>
// Para PZEM-004T: #include <PZEM004Tv30.h>

const char* SSID      = "SUA_REDE_WIFI";
const char* PASSWORD  = "SUA_SENHA_WIFI";
const char* DEVICE_ID = "MONITECH-001";
const char* FW_VER    = "1.0.0";

WebSocketsServer ws(81);
// PZEM004Tv30 pzem(Serial2, 16, 17); // RX=16, TX=17

unsigned long ultimoEnvio = 0;
const int INTERVALO_ENVIO = 2000; // ms

float lerVolts()  { return 0; }
float lerAmps()   { return 0; }
float lerWatts()  { return 0; }
float lerFP()     { return 0; }
float lerHz()     { return 0; }
float lerKwh()    { return 0; }

// Com PZEM-004T real (descomente):
// float lerVolts()  { return pzem.voltage(); }
// float lerAmps()   { return pzem.current(); }
// float lerWatts()  { return pzem.power(); }
// float lerFP()     { return pzem.pf(); }
// float lerHz()     { return pzem.frequency(); }
// float lerKwh()    { return pzem.energy(); }

void enviarDados() {
  StaticJsonDocument<256> doc;
  doc["id"]     = DEVICE_ID;
  doc["fw"]     = FW_VER;
  doc["chip"]   = ESP.getEfuseMac();
  doc["ip"]     = WiFi.localIP().toString();
  doc["ssid"]   = WiFi.SSID();
  doc["rssi"]   = WiFi.RSSI();
  doc["heap"]   = ESP.getFreeHeap();
  doc["sensor"] = "PZEM-004T";
  doc["volts"]  = lerVolts();
  doc["amps"]   = lerAmps();
  doc["watts"]  = lerWatts();
  doc["pf"]     = lerFP();
  doc["hz"]     = lerHz();
  doc["kwh"]    = lerKwh();
  doc["ts"]     = millis();
  String payload;
  serializeJson(doc, payload);
  ws.broadcastTXT(payload);
}

void eventoWS(uint8_t num, WStype_t tipo, uint8_t* payload, size_t tamanho) {
  if (tipo == WStype_CONNECTED) enviarDados();
}

void setup() {
  Serial.begin(115200);
  WiFi.begin(SSID, PASSWORD);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  ws.begin();
  ws.onEvent(eventoWS);
}

void loop() {
  ws.loop();
  if (millis() - ultimoEnvio >= INTERVALO_ENVIO) { enviarDados(); ultimoEnvio = millis(); }
}
Arduino IDE — Dispositivo Monitech com endpoint HTTP GET (polling pelo site) 
#include <WiFi.h>
#include <WebServer.h>
#include <ArduinoJson.h>

const char* SSID      = "SUA_REDE_WIFI";
const char* PASSWORD  = "SUA_SENHA_WIFI";
const char* DEVICE_ID = "MONITECH-001";

WebServer server(80);

float lerVolts()  { return 0; }
float lerAmps()   { return 0; }
float lerWatts()  { return 0; }
float lerFP()     { return 0; }

void handleDados() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  StaticJsonDocument<256> doc;
  doc["id"]    = DEVICE_ID;
  doc["volts"] = lerVolts();
  doc["amps"]  = lerAmps();
  doc["watts"] = lerWatts();
  doc["pf"]    = lerFP();
  doc["hz"]    = 60.0;
  doc["kwh"]   = 12.4 + millis() / 3600000.0;
  doc["rssi"]  = WiFi.RSSI();
  doc["heap"]  = ESP.getFreeHeap();
  doc["ts"]    = millis();
  String json;
  serializeJson(doc, json);
  server.send(200, "application/json", json);
}

void setup() {
  Serial.begin(115200);
  WiFi.begin(SSID, PASSWORD);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  server.on("/api/data", HTTP_GET, handleDados);
  server.begin();
}

void loop() { server.handleClient(); }
Arduino IDE — Dispositivo Monitech como Publisher MQTT 
#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

const char* SSID        = "SUA_REDE_WIFI";
const char* PASSWORD    = "SUA_SENHA_WIFI";
const char* MQTT_SERVER = "broker.hivemq.com";
const int   MQTT_PORT   = 1883;
const char* TOPICO      = "MONITECH/home/energy";
const char* DEVICE_ID   = "MONITECH-001";

WiFiClient   espClient;
PubSubClient mqtt(espClient);
unsigned long ultimaMensagem = 0;

float lerVolts() { return 220.0 + random(-10, 10) * 0.1; }
float lerAmps()  { return 5.0  + random(-5, 5)  * 0.1; }

void reconectar() {
  while (!mqtt.connected()) {
    if (mqtt.connect(DEVICE_ID)) mqtt.publish(TOPICO "/status", "online");
    else delay(3000);
  }
}

void publicarDados() {
  StaticJsonDocument<256> doc;
  float v = lerVolts(), a = lerAmps();
  doc["id"]    = DEVICE_ID;
  doc["volts"] = v; doc["amps"] = a;
  doc["watts"] = v * a * 0.85;
  doc["pf"]    = 0.85; doc["hz"] = 60.0;
  doc["kwh"]   = 12.4 + millis() / 3600000.0;
  doc["rssi"]  = WiFi.RSSI(); doc["ts"] = millis();
  char buf[256];
  serializeJson(doc, buf);
  mqtt.publish(TOPICO, buf);
}

void setup() {
  Serial.begin(115200);
  WiFi.begin(SSID, PASSWORD);
  while (WiFi.status() != WL_CONNECTED) { delay(500); }
  mqtt.setServer(MQTT_SERVER, MQTT_PORT);
}

void loop() {
  if (!mqtt.connected()) reconectar();
  mqtt.loop();
  if (millis() - ultimaMensagem > 2000) { publicarDados(); ultimaMensagem = millis(); }
}

MONITECH é uma plataforma de monitoramento de energia elétrica em tempo real, desenvolvida para residências e pequenos comércios.

Com suporte a múltiplos protocolos (WebSocket, HTTP, MQTT e Serial/USB) e sensores como o PZEM-004T e ACS712, oferece análise detalhada de consumo, tensão, corrente e fator de potência.

AVISO IMPORTANTE
Para receber leituras de energia em tempo real, você precisa conectar um dispositivo de medição Dispositivo Monitech ao seu painel. Ele irá monitorar tensão, corrente, potência e consumo da sua residência continuamente.
Como configurar o dispositivo:
  • Monte um Dispositivo Monitech com o sensor de medição PZEM-004T
  • Conecte o dispositivo à sua rede Wi-Fi doméstica
  • Acesse a aba "Dispositivo MONITECH" no painel
  • Escolha o protocolo de comunicação: WebSocket, HTTP ou MQTT
  • Insira o IP do Dispositivo Monitech e clique em CONECTAR
